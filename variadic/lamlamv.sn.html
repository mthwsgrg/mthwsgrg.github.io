Lemma step_betav' (p:scope) (n: nat) b (tBar: vec n (tm p)) t':
  t' = b[scons_n n tBar ids] → step (appv _ (lamv n b) tBar) t'.
Proof. intros →. now constructor. Qed.

Lemma step_beta' p b (t t': tm p):
  t' = b[t..] → step (app (lam b) t) t'.
Proof. intros →. now constructor. Qed.

Multi-step reduction

Lemma mstep_lamv (n: nat) (p:scope) (s t : tm (n ± p)) :
  star step s t → star step (lamv n s) (lamv n t).
Proof. induction 1; eauto. Qed.

Lemma mstep_lam p (s t : tm (Sp p)) :
  star step s t → star step (lam s) (lam t).
Proof. induction 1; eauto. Qed.

Lemma mstep_app p (s1 s2 : tm p) (t1 t2 : tm p) :
  star step s1 s2 → star step t1 t2 → star step (app s1 t1) (app s2 t2).
Proof with eauto.
  intros ms. induction 1. induction ms... auto...
Qed.

Substitutivity of reduction
Lemma step_inst {p q:scope} (σ : fin p → tm q) (s t : tm p) :
  step s t → step s[σ] t[σ].
Proof.
  intros st. revert q σ. induction st; intros; cbn.
  - apply step_betav'. now asimpl.
  - apply step_absv. eapply IHst.
  - apply step_appvL, IHst.
  - apply step_beta'. now asimpl.
  - apply step_abs. eapply IHst.
  - apply step_appL, IHst.
  - apply step_appR, IHst.
Qed.

Lemma mstep_inst (p q:scope) (σ : fin p → tm q) (s t : tm p) :
  star step s t → star step s[σ] t[σ].
Proof. induction 1; eauto using step_inst. Qed.

Anti-renaming lemma
Lemma step_naturality (p q:scope) (s: tm p) (ξ: fin p → fin q) t :
  step s⟨ξ⟩ t → ∃ t', step s t' ∧ t = t'⟨ξ⟩.
Proof.
  intros H.
  dependent induction H.
  - destruct s; try inversion x. subst.

    apply inj_pair2_eq_dec in H2; try (apply eq_nat_dec).
    rewrite H2 in x.
    destruct s; try inversion H1. subst.
    simpl in H1.
    rewrite H1 in x.
    apply inj_pair2_eq_dec in H3; try (apply eq_nat_dec).
    rewrite H3.
    ∃ (subst_tm (scons_n n v ids) s). split. eauto.
    asimpl. reflexivity.

  - destruct s; try inversion x. subst.
    apply inj_pair2_eq_dec in H2; try (apply eq_nat_dec).
    rewrite H2 in H.
    rewrite H2 in IHstep.
    edestruct (IHstep _ s) as (?&?&?);[reflexivity|].
    subst. eexists;eauto.

  - destruct s; inversion x. subst.
    apply inj_pair2_eq_dec in H3; try (apply eq_nat_dec).
    edestruct (IHstep _ s) as (?&?&?); [reflexivity|].
    subst. eexists; eauto.

  - destruct s; try inversion x. subst.
    destruct s1; try inversion H0. subst.
    ∃ (subst_tm (s2..) s1). split. eauto.
    asimpl. reflexivity.
  - destruct s; inversion x. subst.
    edestruct (IHstep _ s) as (?&?&?); [reflexivity|].
    subst. eexists; eauto.
  - destruct s; inversion x. subst.
    edestruct (IHstep _ s3) as (?&?&?); [reflexivity|].
    subst. eexists; eauto.
  - destruct s; inversion x. subst.
    edestruct (IHstep _ s2) as (?&?&?); [reflexivity|].
    subst. eexists; eauto.
Qed.

Definition value := fun {p : scope} (s : tm p) ⇒ match s with
                            | lamv _ _ ⇒ True
                            | lam _ ⇒ True
                            | _ ⇒ False
                            end.

Lemma eq_nat: ∀ (n m:nat), {n=m} + {n≠m}.
  decide equality.
Qed.

Schäfer's expression relation
Inductive E_strong' {p:scope} (L_A : tm p → Prop) (s : tm p) : Prop :=
 | E_strong_In : (value s → L_A s) → (∀ s', step s s' → E_strong' L_A s') → E_strong' L_A s.

Value relation
Fixpoint L_strong {p:scope} (A : ty): tm p → Prop.
  refine ( match A with
  | B ⇒ fun _ ⇒ True
  | @arr n F A2 ⇒ fun e ⇒ match e with
                            | (lamv n' s) ⇒ match (eq_nat n n') with
                                            | left _ equ ⇒ _
                                            | right _ ⇒ False
                                          end
                            | _ ⇒ False
                           end
  | Fun A1 A2 ⇒ fun e ⇒ match e with
                      | (lam s) ⇒ ∀ q (ξ: fin p → fin q) v, E_strong' (L_strong q A1) v → E_strong' (L_strong q A2) (subst_tm (v .: (ξ >> var_tm)) s)
                      | _ ⇒ False
                      end
  end).

  rewrite <- equ in s.
  exact (∀ (q:scope) (ξ: fin p → fin q) (vs: vec n (tm q)), (∀ i, E_strong' (L_strong q (F i)) (vs i)) → E_strong' (L_strong q A2) (subst_tm (scons_n n vs (ξ >> var_tm)) s)).

Defined.

Definition E_strong {p:scope} (A : ty) : tm p → Prop :=
  E_strong' (L_strong A).

Lemma E_strong_sn {p: scope} (s : tm p) A:
  E_strong A s → sn step s.
Proof.
induction 1. constructor. intros s' HH. eauto.
Qed.

Lemma E_strong_step {p:scope} (s: tm p) L_A :
  E_strong' L_A s → ∀ s', step s s' → E_strong' L_A s'.
Proof.
 induction 1; eauto.
Qed.

Lemma E_strong_base {p:scope} (s: tm p) A :
  value s → E_strong A s → L_strong A s.
Proof.
  intros C []; eauto.
Qed.

Lemma E_strong_var {p:scope} A (x: fin p) :
  E_strong A (var_tm x).
Proof.
  constructor; eauto.
  - cbn. contradiction.
  - intros. inversion H.
Qed.

Context relation
Definition G_strong {p q:scope} (Γ : ctx p) : (fin p → tm q) → Prop :=
  fun σ ⇒ ∀ (x : fin p), E_strong (Γ x) (σ x) .

Semantic typing
Definition has_ty_sem_strong {p:scope} (Γ : ctx p) (s: tm p) (A: ty) : Prop :=
  ∀ q (σ: fin p → tm q), G_strong Γ σ → E_strong A (subst_tm σ s).

Lemma L_close_ren_lam (p q:scope) (ξ: fin p → fin q) s A :
  L_strong A s → L_strong A s⟨ξ⟩.
Proof.
  induction A; eauto.
 - destruct s; try contradiction.
    subst. intros. cbn in ×. intros. asimpl.
    destruct eq_nat; eauto.
      generalize s H H0 IHA.
      rewrite <- e.
      clear H IHA H0 s.
      intros.
      unfold eq_rect_r in ×.
      unfold eq_rect in ×.
      simpl in ×.
      asimpl; eauto.
  - destruct s; try contradiction.
    subst. asimpl. intros. cbn in ×. intros. asimpl. eauto.
Qed.

Lemma close_ren :
  (∀ (p q:scope) (ξ: fin p → fin q) s A, E_strong A s → E_strong A s⟨ξ⟩).
Proof.
  intros. induction H.
  unfold E_strong in H1.
  constructor.
  intros.
  assert (value s) by (now destruct s).
  specialize (H H3). now apply L_close_ren_lam.
  intros.
  apply step_naturality in H2.
  destruct H2 as (?&?&?); subst.
  apply H1; eauto.
Qed.

Lemma value_anti_renaming_lam (p q:scope) s (ξ: fin p → fin q) :
  value s⟨ξ⟩ → value s.
Proof.
  destruct s; eauto.
Qed.

Lemma helper_lam: ∀ p q (s: tm (⥠ + q)) (σ: fin p → tm q) A B Γ, sn step s → G_strong Γ σ → (∀ q' (τ: fin (⥠ + q) → tm q'), G_strong (scons A Γ) ((up_tm_tm σ) >> [τ]) → (E_strong B s[τ] )) → E_strong (Fun A B) (lam s).
  intros.
  induction H. constructor.
  intros V q'.
  intros.
  apply H1. asimpl.
  unfold G_strong.
  intros.
  destruct x0.
  - asimpl; simpl.
     asimpl.
     renamify.
     apply close_ren. unfold G_strong in H0. eauto.
  - asimpl; simpl.
    unfold E_strong.
    eauto.

  - intros. inv H3.
      apply H2; eauto.
      intros. specialize (H1 _ _ H4). eapply E_strong_step; eauto.
      eauto using step_inst.
Qed.

Lemma helper_lamv : ∀ p q n (s: tm (n ± q)) (σ: fin p → tm q) A B Γ, sn step s → G_strong Γ σ → (∀ q' (τ: fin ( n ± q) → tm q'), G_strong (scons_n n A Γ) ((upList_tm_tm n σ) >> [τ]) → (E_strong B s[τ] )) → E_strong (@arr n A B) (lamv n s).
  intros.
  induction H; constructor.
      intros V.
      cbn; destruct (eq_nat n n);eauto; subst.
      intro q'.
      intros.
      unfold eq_rect_r. unfold eq_rect.
      assert (e = eq_refl).
      apply Eqdep_dec.eq_proofs_unicity.
      decide equality.
      subst. simpl.
      apply H1. asimpl.
      unfold G_strong.
     intros.
     destruct x0.
      × asimpl; simpl.
         unfold E_strong.
         pose proof (H3 v).
         eauto.
      × asimpl; simpl.
         asimpl.
         renamify.
         apply close_ren. unfold G_strong in H0. eauto.
   × intros. inv H3.
      apply H2; eauto.
      intros. specialize (H1 _ _ H4). eapply E_strong_step; eauto.
      eauto using step_inst.
Qed.

Lemma sn_fundamental p Gamma (s: tm p) A :
  has_type Gamma s A → has_ty_sem_strong Gamma s A.
Proof.
  intros C. induction C as [p | p Γ n A B s H IH | p Γ n A B s t H IH H' IH'

   | p Γ A B s H IH | p Γ A B s t H IH H' IH' ].

  - intros q sigma G. apply G.
  - intros q sigma G. cbn.
    assert (sn step (subst_tm (upList_tm_tm n sigma) s)).
    { eapply E_strong_sn. unfold has_ty_sem_strong in IH. apply IH.
      intro.
      unfold E_strong.
      destruct x.
      + asimpl; simpl.
        apply E_strong_var.
      + asimpl; simpl.
        apply close_ren. eauto.
    }
    remember (subst_tm (upList_tm_tm n sigma) s) as s'.
    unfold has_ty_sem_strong in IH.
    assert (∀ q' (tau : fin (n ± q) → tm q'), G_strong (scons_n n A Γ) (upList_tm_tm n sigma >> [tau]) → E_strong B s'[tau] ).
    {
      intros. rewrite Heqs'. revert H1. asimpl. eauto.
    }
    eapply helper_lamv; eauto.
  - intros q sigma G.
    specialize (IH _ _ G). unfold has_ty_sem_strong in IH'.
    asimpl.
    remember (subst_tm sigma s) as s'.
    remember (vec_map (subst_tm sigma) t) as t'.
    apply E_strong_sn in IH as IH1'.
    assert (∀ x:vindex n, E_strong (A x) (t' x)) as IHx.
    unfold vec_map in Heqt'.
    rewrite Heqt'.
    eauto.
    clear s Heqs' Heqt' H H'.
    revert t' IH' IHx. induction IH1' as [s].
    intros t' IH2'.
    constructor.
    + contradiction.
    + intros ? HH. inv HH; eauto using E_strong_step.
      × apply E_strong_base in IH.
        -- cbn in IH.
           destruct (eq_nat _ _).
           ** unfold eq_rect_r in IH.
               unfold eq_rect in IH.
               assert (e = eq_refl).
               apply Eqdep_dec.eq_proofs_unicity.
               decide equality. subst. simpl in IH.
               specialize (IH _ id t'). eapply IH.
               unfold E_strong in IH2'.
               eauto.
           ** destruct IH.
       -- reflexivity.
      × apply H0; eauto. eapply E_strong_step; eauto.
  - intros q sigma G. cbn.
      assert (sn step (subst_tm (up_tm_tm sigma) s)).
      eapply E_strong_sn. unfold has_ty_sem_strong in IH. apply IH.
      intro.
      unfold E_strong.
      destruct x.
      + asimpl; simpl.
        apply close_ren. eauto.
      + asimpl; simpl.
        apply E_strong_var.
   + remember (subst_tm (up_tm_tm sigma) s) as s'.
    assert (∀ q' (tau : fin (Sp q) → tm q'), G_strong (scons A Γ) (up_tm_tm sigma >> [tau]) → E_strong B s'[tau]).
    intros. rewrite Heqs'. revert H1. asimpl. eauto.
    eapply helper_lam; eauto.

  - intros q sigma D.
    specialize (IH _ _ D). specialize (IH' _ _ D).
    cbn.
    remember (subst_tm sigma s) as s'. remember (subst_tm sigma t) as t'.
    apply E_strong_sn in IH as IH1'. apply E_strong_sn in IH' as IH2'.
    clear s t Heqs' Heqt' H H'.
    revert t' IH' IH2'. induction IH1' as [s].
    intros t' IH2 IH2'. induction IH2' as [t].
    constructor.
    + contradiction.
    + intros ? HH. inv HH; eauto using E_strong_step.
      × apply E_strong_base in IH.
       -- cbn in IH.
          specialize (IH _ id t). eapply IH.
        assumption.
       -- reflexivity.
     × apply H0; eauto. eapply E_strong_step; eauto.
     × apply H2; eauto. eapply E_strong_step; eauto.
Qed.

Theorem sn {p:scope} (Γ : fin p → ty) (s: tm p) A:
  has_type Γ s A → sn step s.
Proof.
  intros H%sn_fundamental.
  unfold has_ty_sem_strong in H.
  replace s with (subst_tm ids s) by (now asimpl).
  eapply E_strong_sn. eapply H.
  unfold G_strong.
  intros.
  unfold ids.
  unfold VarInstance_tm.
apply E_strong_var.
Qed.
