Definition value := fun {p : scope} (s : tm p) ⇒ match s with

Logical relation for λλₙ-expressions
Fixpoint L_strong {p:scope} (A : ty): tm p → Prop.
  refine ( match A with
  | B ⇒ fun _ ⇒ True
  | @arr n F A2 ⇒ fun e ⇒ match e with
                            | (lamv n' s) ⇒ match (eq_or_not n n') with
                                            | left _ equ ⇒ _
                                            | right _ ⇒ False
                                          end
                            | _ ⇒ False
                           end
  | Fun A1 A2 ⇒ fun e ⇒ match e with
                      | (lam s) ⇒ ∀ q (ξ: fin p → fin q) v, E_strong' (L_strong q A1) v → E_strong' (L_strong q A2) (subst_tm (v .: (ξ >> var_tm)) s)
                      | _ ⇒ False
                      end
  end).

  rewrite <- equ in s.
  exact (∀ (q:scope) (ξ: fin p → fin q) (vs: vec n (tm q)), (∀ i, E_strong' (L_strong q (F i)) (vs i)) → E_strong' (L_strong q A2) (subst_tm (scons_n n vs (ξ >> var_tm)) s)).

Definition E_strong {p:scope} (A : ty) : tm p → Prop :=
  E_strong' (L_strong A).

Definition G_strong {p q:scope} (Γ : ctx p) : (fin p → tm q) → Prop :=
  fun σ ⇒ ∀ (x : fin p), E_strong (Γ x) (σ x) .

Definition has_ty_sem_strong {p:scope} (Γ : ctx p) (s: tm p) (A: ty) : Prop :=
  ∀ q (σ: fin p → tm q), G_strong Γ σ → E_strong A (subst_tm σ s).

Lemma L_close_ren_lam (p q:scope) (ξ: fin p → fin q) s A :
  L_strong A s → L_strong A (ren_tm ξ s).
Proof.
  induction A; eauto.
 - destruct s; try contradiction.
    subst. intros. cbn in ×. intros. asimpl.
    destruct eq_or_not; eauto.
      generalize s H H0 IHA.
      rewrite <- e.
      clear H IHA H0 s.
      intros.
      unfold eq_rect_r in ×.
      unfold eq_rect in ×.
      simpl in ×.
      asimpl; eauto.
  - destruct s; try contradiction.
    subst. asimpl. intros. cbn in ×. intros. asimpl. eauto.
Qed.

Lemma close_ren :

  intros.
  apply step_naturality in H2.
  destruct H2 as (?&?&?); subst.
  apply H1; eauto.
Qed.

Lemma helper_lam: ∀ p q (x: tm (Sp q)) (σ: fin p → tm q) A B Γ, sn step x → G_strong Γ σ → (∀ q' (τ: fin (Sp q) → tm q'), G_strong (scons A Γ) ((up_tm_tm σ) >> subst_tm τ) → (E_strong B (subst_tm τ x) )) → E_strong (Fun A B) (lam x).
  intros.
  induction H. constructor.
  intros V q'.
  intros.
  apply H1. asimpl.
  unfold G_strong.
  intros.
  destruct x0.
  - asimpl; simpl.
     asimpl.
     renamify.
     apply close_ren. unfold G_strong in H0. eauto.
  - asimpl; simpl.
    unfold E_strong.
    eauto.

  - intros. inv H3.
      apply H2; eauto.
      intros. specialize (H1 _ _ H4). eapply E_strong_step; eauto.
      eauto using step_inst.
Qed.

Lemma helper_lamv : ∀ p q n (x: tm (n ± q)) (σ: fin p → tm q) A B Γ, sn step x → G_strong Γ σ → (∀ q' (τ: fin ( n ± q) → tm q'), G_strong (scons_n n A Γ) ((upList_tm_tm n σ) >> [τ]) → (E_strong B x[τ] )) → E_strong (@arr n A B) (lamv n x).

  intros.
  induction H; constructor.
      intros V.
      cbn; destruct (eq_or_not n n);eauto; subst.
      intro q'.
      intros.
      unfold eq_rect_r. unfold eq_rect.
      assert (e = eq_refl).
      apply Eqdep_dec.eq_proofs_unicity.
      decide equality.
      subst. simpl.
      apply H1. asimpl.
      unfold G_strong.
     intros.
     destruct x0.
      × asimpl; simpl.
         unfold E_strong.
         pose proof (H3 v).
         eauto.
      × asimpl; simpl.
         asimpl.
         renamify.
         apply close_ren. unfold G_strong in H0. eauto.

   × intros. inv H3.
      apply H2; eauto.
      intros. specialize (H1 _ _ H4). eapply E_strong_step; eauto.
      eauto using step_inst.
Qed.

Lemma sn_fundamental p Gamma (s: tm p) A :
  has_type Gamma s A → has_ty_sem_strong Gamma s A.
Proof.
  intros C. induction C as [p | p Γ n A B s H IH | p Γ n A B s t H IH H' IH'

   | p Γ A B s H IH | p Γ A B s t H IH H' IH' ].

  - unfold has_ty_sem_strong. intros. cbn. apply H.
  - intros q sigma ctx. cbn.
    assert (sn step (subst_tm (upList_tm_tm n sigma) s)).

      eapply E_strong_sn. unfold has_ty_sem_strong in IH. apply IH.
      intro.
      unfold E_strong.
      destruct x.
      + asimpl; simpl.
        apply E_strong_var.
      + asimpl; simpl.
        apply close_ren. eauto.

   + remember (subst_tm (upList_tm_tm n sigma) s) as s'.
    unfold has_ty_sem_strong in IH.
    assert (∀ q' (tau : fin (n ± q) → tm q'), G_strong (scons_n n A Γ) (upList_tm_tm n sigma >> [tau]) → E_strong B (s'[tau])).
    { intros. rewrite Heqs'. revert H1. asimpl. eauto. }
    clear Heqs' IH H s.
    eapply helper_lamv; eauto.

  - intros q sigma D.
    specialize (IH _ _ D). unfold has_ty_sem_strong in IH'.
    assert (∀ (x:vindex n), E_strong (A x) (subst_tm sigma (t x))).
    eauto.
    clear IH'.
    remember H0 as IH'.
    clear HeqIH' H0.
    cbn.
    remember (subst_tm sigma s) as s'.
    remember (vec_map ( subst_tm sigma ) t) as t'.
    assert (∀ x : vindex n, E_strong (A x) (subst_tm sigma (t x))) as _IH.
    assumption.
    clear IH'.
    assert (∀ x:vindex n, E_strong (A x) (t' x)) as IH'.
    unfold vec_map in Heqt'.
    rewrite Heqt'.
    assumption.
    clear _IH.
    assert (sn step s') as IH1'.
    apply E_strong_sn in IH.
    assumption.
    assert (∀ x, sn step (t' x)) as IH2'.
    intros.
    pose proof (IH' x).
    apply E_strong_sn in H0.
    assumption.
    clear s t Heqs' Heqt' H H'.
    revert t' IH' IH2'. induction IH1' as [s].
    intros t' IH2 IH2'.
     constructor.
    + contradiction.
    + intros ? HH. inv HH; eauto using E_strong_step.
      × apply E_strong_base in IH.
        -- cbn in IH.
           destruct (eq_or_not n n).

           ** unfold eq_rect_r in IH.
               unfold eq_rect in IH.
               assert (e = eq_refl).
               apply Eqdep_dec.eq_proofs_unicity.
               decide equality. subst. simpl in IH.
              specialize (IH _ id t'). eapply IH.
              eauto.
           ** destruct IH.
       -- reflexivity.
      × apply H0; eauto. eapply E_strong_step; eauto.

  - intros q sigma ctx. cbn.
      assert (sn step (s [(up_tm_tm sigma)])).
      eapply E_strong_sn. unfold has_ty_sem_strong in IH. apply IH.
      intro.
      unfold E_strong.
      destruct x.
      + asimpl; simpl.
        apply close_ren. eauto.
      + asimpl; simpl.
        apply E_strong_var.
      + remember ( s[up_tm_tm sigma]) as s'.
    unfold has_ty_sem_strong in IH.
    assert (∀ q' (tau : fin (⥠+ q) → tm q'), G_strong (scons A Γ) (up_tm_tm sigma >> [tau]) → E_strong B ( s'[tau])).
    intros. rewrite Heqs'. revert H1. asimpl. eauto.
    asimpl in ×. rewrite <- Heqs'.
    clear Heqs' IH H s.
    eapply helper_lam; eauto.

  - intros q sigma D.
    specialize (IH _ _ D). specialize (IH' _ _ D).
    cbn.
    remember (subst_tm sigma s) as s'. remember (subst_tm sigma t) as t'.
    apply E_strong_sn in IH as IH1'. apply E_strong_sn in IH' as IH2'.
    clear s t Heqs' Heqt' H H'.
    revert t' IH' IH2'. induction IH1' as [s].
    intros t' IH2 IH2'. induction IH2' as [t].
    constructor.
    + contradiction.
    + intros ? HH. inv HH; eauto using E_strong_step.
      × apply E_strong_base in IH.
       -- cbn in IH.
          specialize (IH _ id t). eapply IH.
        assumption.
       -- reflexivity.
     × apply H0; eauto. eapply E_strong_step; eauto.
     × apply H2; eauto. eapply E_strong_step; eauto.
Qed.


Theorem sn {p:scope} (Γ : fin p → ty) (s: tm p) A:
  has_type Γ s A → sn step s.
Proof.
  intros H%sn_fundamental.
  unfold has_ty_sem_strong in H.
  replace s with (subst_tm ids s) by (now asimpl).
  eapply E_strong_sn. eapply H.
  unfold G_strong.
  intros.
  unfold ids.
  unfold VarInstance_tm.
apply E_strong_var.
Qed.
