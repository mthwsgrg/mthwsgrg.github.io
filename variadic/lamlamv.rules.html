Reduction rules/operational semantics
Inductive step {p:scope} : tm p → tm p → Prop :=
| step_betav (n: nat) b (t: vec n (tm p) ) : step (appv _ (lamv n b) t) (subst1 (scons_n n t ids) b)
| step_absv (n: nat) b1 b2 : @step (n ± p) b1 b2 → step (lamv n b1) (lamv n b2)
| step_appvL (n: nat) s1 s2 t : step s1 s2 → step (appv n s1 t) (appv _ s2 t)
| step_beta b t : step (app (lam b) t) b[t..]
| step_abs b1 b2 : @step (Sp p) b1 b2 → step (lam b1) (lam b2)
| step_appL s1 s2 t : step s1 s2 → step (app s1 t) (app s2 t)
| step_appR s t1 t2 : step t1 t2 → step (app s t1) (app s t2).

#[export] Hint Constructors step.

Inductive ty : Type :=
  | B : ty
  | arr n : (vec n ty) → ty → ty
  | Fun : ty → ty → ty .

Definition ctx (p:scope) := fin p → ty.

Typing rules
Inductive has_type {p:scope} (Γ : ctx p) : tm p → ty → Prop :=
| ty_var (x : fin p) :
    has_type Γ (var_tm x) (Γ x)
| ty_absv (n:nat) (ABar : vec n ty) ( A : ty) (s : tm (n ± p)) :
    @has_type (n ± p) (scons_n n ABar Γ) s A →
    has_type Γ (lamv n s) (arr ABar A)
| ty_appv (n:nat) (ABar : vec n ty) ( B : ty) (s : tm p) tBar :
    has_type Γ s (arr ABar B) →
    (∀ x, has_type Γ (tBar x) (ABar x)) →
    has_type Γ (appv n s tBar) B
| ty_abs (A B : ty) (s : tm (Sp p)) :
    @has_type (Sp p) (A .: Γ) s B →
    has_type Γ (lam s) (Fun A B)
| ty_app (A B : ty) (s t : tm p) :
    has_type Γ s (Fun A B) →
    has_type Γ t A →
    has_type Γ (app s t) B.

Notation "Gamma |- s : A" := (has_type Gamma s A) (at level 20, s at level 99).

#[export] Hint Constructors has_type.
