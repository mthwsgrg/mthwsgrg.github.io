Definition ap {X Y} (f : X → Y) {x y : X} (p : x = y) : f x = f y :=
  match p with eq_refl ⇒ eq_refl end.

Definition apc {X Y} {f g : X → Y} {x y : X} (p : f = g) (q : x = y) : f x = g y :=
  match q with eq_refl ⇒ match p with eq_refl ⇒ eq_refl end end.

Fixpoint fin (p : scope) : Type :=
  match p with
  | 0 ⇒ False
  | ⥠ + p' ⇒ option (fin p')
  | n ± p' ⇒ (vindex n) + (fin p')
  end.

Definition zero {p: scope} : fin (⥠ + p) := None.
Definition zero_n {n: nat} {p : scope}
                  (i: vindex n) : fin (n ± p) := inl i.

Definition shift {p: scope} (x: fin p) : fin (⥠ + p) := Some x.
Definition shift_n {p: scope} (n: nat) (x: fin p) : fin (n ± p) := inr x.

Definition scons {A: Type} {p: scope} (a : A)
  (f : fin p → A) (x: fin (⥠ + p)) : A :=
 match x with
 | None ⇒ a
 | Some v ⇒ f v
 end.

Definition scons_n {A: Type} {n : nat} {p:scope} (v : vec n A) (f : fin p → A) (x: fin (n ± p)) : A :=
  match x with
   | inl o ⇒ v o
   | inr o ⇒ f o
  end.

Proofs for substitution primitives

Forward Function Composition Substitutions represented as functions are ubiquitious in this development and we often have to compose them, without talking about their pointwise behaviour. That is, we are interested in the forward compostion of functions, f o g, for which we introduce a convenient notation, "f >> g". The direction of the arrow serves as a reminder of the forward nature of this composition, that is first apply f, then g.

Arguments funcomp {X Y Z} (g)%fscope (f)%fscope.

Notation "f >> g" := (funcomp g f) (at level 50).
Open Scope subst_scope.

Notation "x .: f" := (@scons _ _ x f) (at level 55) : subst_scope.

Generic lifting operation for renamings
Definition up_ren m n (xi : ren m n) : ren (⥠ + m) (⥠ + n) :=
  zero .: xi >> shift.

Generic proof that lifting of renamings composes.
Lemma up_ren_ren p q r (xi: ren p q) (zeta : ren q r) (rho: ren p r) (E: ∀ x, (xi >> zeta) x = rho x) :
  ∀ x, (up_ren xi >> up_ren zeta) x = up_ren rho x.
Proof.
  intros x.
  destruct x.
  - unfold up_ren. simpl in ×.
     unfold funcomp. simpl in ×.
     specialize (E f).
     unfold funcomp in E. rewrite E. reflexivity.
  - unfold funcomp.
    reflexivity.
Qed.

Arguments up_ren_ren {p q r} xi zeta rho E.

Lemma scons_eta {T} {p : scope} (f : fin (⥠ + p) → T) :
  f zero .: shift >> f = f.

  - reflexivity.

Variadic Substitution Primitives


Lemma scons_n_head' {A} {n:nat} {p:scope} (v : vec n A) (g : fin p → A) i:
  (scons_n v g) (zero_n i) = v i.
Proof.
  reflexivity.
Qed.

Lemma scons_n_head A (n:nat) {p:scope} (v : vec n A) (g : fin p → A) :
  (zero_n >> scons_n v g) = v.
Proof. fext. intros z. unfold funcomp. apply scons_n_head'. Qed.

Lemma scons_n_tail' A (n:nat) {p:scope} (v : vec n A) (g : fin p → A) x :
  scons_n v g (shift_n n x) = g x.
Proof. reflexivity. Qed.

Lemma scons_n_tail A (n:nat) {p:scope} (v : vec n A) (g : fin p → A) :
  shift_n n >> scons_n v g = g.
Proof. fext. intros z. unfold funcomp. apply scons_n_tail'. Qed.

Lemma scons_n_comp' A B (n:nat) {p:scope} (v : vec n A) (g : fin p → A) (h : A → B) x:
 h (scons_n v g x) = scons_n (v >> h) (g >> h) x.
Proof.
  simpl in ×.
  destruct x.
  - simpl. reflexivity.
  - simpl. reflexivity.
Qed.

Lemma scons_n_comp {A B} {n:nat} {p:scope} {v : vec n A} {g : fin p → A} {h : A → B} :
  (scons_n v g) >> h = scons_n (v >> h) (g >> h).
Proof. fext. intros z. unfold funcomp. apply scons_n_comp'. Qed.

Lemma scons_n_congr {A} {n:nat} {p:scope} (v v' : vec n A) (g g': fin p → A) x:
  (∀ i, v i = v' i) → (∀ x, g x = g' x) → scons_n v g x = scons_n v' g' x.
Proof.
  intros.
  simpl in ×.
  destruct x.
  - eauto.
  - eauto.
Qed.

Generic n-ary lifting operation.
Definition upRen_n n { p : scope } { q : scope } (xi : fin (p) → fin (q)) : fin ( n ± p) → fin (n ± q) :=
   scons_n (zero_n ) (xi >> shift_n _).

Arguments upRen_n n {p q} xi.

Generic proof for composition of n-ary lifting.
Lemma up_ren_ren_n {n:nat} {p q r:scope} (xi: ren p q) (zeta : ren q r) (rho: ren p r) (E: ∀ x, (xi >> zeta) x = rho x) :
  ∀ x, (upRen_n n xi >> upRen_n n zeta) x = upRen_n n rho x.
Proof.
  intros x. destruct x.
  - unfold upRen_n. unfold funcomp. now simpl.
  - unfold upRen_n. unfold funcomp. simpl.
    now rewrite <- E.
Qed.

Arguments zero_n n {p}.
Arguments scons_n {A} n {p} v f.

Lemma scons_n_eta {A} {n:nat} {p:scope} {v : vec n A}
      {g : fin p → A} (h: fin (n ± p) → A) {z: fin (n ± p)}:
  (∀ x, g x = h (shift_n n x)) → (∀ i, v i = h (zero_n n i)) → scons_n n v g z = h z.
Proof.
  intros H1 H2. destruct z.
  - simpl. eauto.
  - simpl. eauto.
Qed.

Tactics for Scoped Syntax

Generic fsimpl tactic: simplifies the above primitives in a goal.
Ltac fsimpl :=
  repeat match goal with
         | [|- context[id >> ?f]] ⇒ change (id >> f) with f
         | [|- context[?f >> id]] ⇒ change (f >> id) with f
         | [|- context [id ?s]] ⇒ change (id s) with s
         | [|- context[comp ?f ?g]] ⇒ change (comp f g) with (g >> f)
         | [|- context[(?f >> ?g) >> ?h]] ⇒
           change ((f >> g) >> h) with (f >> (g >> h))

         | [|- zero_n >> scons_n ?f ?g] ⇒ rewrite scons_n_head

         | [|- context[(?s.:?sigma) zero]] ⇒ change ((s.:sigma) zero) with s
         | [|- context[(?s.:?sigma) (shift ?m)]] ⇒ change ((s.:sigma) (shift m)) with (sigma m)

         | [|- context[idren >> ?f]] ⇒ change (idren >> f) with f
         | [|- context[?f >> idren]] ⇒ change (f >> idren) with f
         | [|- context[?f >> (?x .: ?g)]] ⇒ change (f >> (x .: g)) with g
         | [|- context[?x2 .: shift >> ?f]] ⇒ change x2 with (f zero); rewrite (@scons_eta _ _ f)
         | [|- context[?f zero .: ?g]] ⇒ change g with (shift >> f); rewrite scons_eta

         |[|- _ = ?h (?f ?s)] ⇒ change (h (f s)) with ((f >> h) s)
         |[|- ?h (?f ?s) = _] ⇒ change (h (f s)) with ((f >> h) s)

         | _ ⇒ first [progress (rewrite scons_comp) | progress (rewrite scons_eta_id) | progress (autorewrite with FunctorInstances)]
         end.

Generic fsimpl tactic: simplifies the above primitives in the context
Ltac fsimplc :=
  repeat match goal with
         | [H: context[id >> ?f] |- _] ⇒ change (id >> f) with f in H
         | [H: context[?f >> id]|- _] ⇒ change (f >> id) with f in H
         | [H: context [id ?s]|- _] ⇒ change (id s) with s in H
         | [H: context[comp ?f ?g]|- _] ⇒ change (comp f g) with (g >> f) in H
         | [H: context[(?f >> ?g) >> ?h]|- _] ⇒
           change ((f >> g) >> h) with (f >> (g >> h)) in H
         | [H: context[(?s.:?sigma) zero]|- _] ⇒ change ((s.:sigma) zero) with s in H
         | [H: context[(?s.:?sigma) zero]|- _] ⇒ change ((s.:sigma) zero) with s in H
         | [H: context[(?s.:?sigma) (shift ?m)]|- _] ⇒ change ((s.:sigma) (shift m)) with (sigma m) in H
                                                                                      |[H : context[ _ = ?h (?f ?s)]|- _] ⇒ change (h (f s)) with ((f >> h) s) in H
         |[H: context[?h (?f ?s) = _]|- _] ⇒ change (h (f s)) with ((f >> h) s) in H
         | [H: context[idren >> ?f]|- _] ⇒ change (idren >> f) with f in H
         | [H: context[?f >> idren]|- _] ⇒ change (f >> idren) with f in H
         | [H: context[?f >> (?x .: ?g)]|- _] ⇒
           change (f >> (x .: g)) with g in H
         | [H: context[?x2 .: shift >> ?f]|- _] ⇒
           change x2 with (f zero) in H; rewrite (@scons_eta _ _ f) in H
         | [H: context[?f zero .: ?g]|- _] ⇒
           change g with (shift >> f) in H; rewrite scons_eta in H
         | _ ⇒ first [progress (rewrite scons_comp in *) | progress (rewrite scons_eta_id in *) | progress (autorewrite with FunctorInstances in *)]
         end.

Takes a vindex and add n to it

Fixpoint vindex_add (n : nat) {m} : vindex m → vindex (n + m) :=
  fun m ⇒ match n with
        | O ⇒ m
        | S n ⇒ Some (vindex_add n m)
        end.

Takes a vindex and returns the vindex but in an expanded domain with m more elements

Definition vindex_expand {n : nat} {m} : vindex n → vindex (n + m).
Proof.
  induction n.
  - intros [].
  - intros [x|].
     + exact (vindex_add 1 (IHn x)).
     + exact None.
Defined.

A vindex from a domain with "m+n"

Lemma destruct_vindex {m n} (x : vindex (m + n)):
  (∃ x', x = vindex_expand x') ∨ ∃ x', x = vindex_add m x'.
Proof.
  induction m; simpl in ×.
  - right. eauto.
  - destruct x as [x|].
    + destruct (IHm x) as [[x' ->] |[x' ->]].
      × left. now ∃ (Some x').
      × right. eauto.
    + left. ∃ None. eauto.
Qed.

Vector operations
Definition vcons {A : Type} {n : nat} (x : A) (v : vec n A) (i : vindex ( (S n))) : A :=
  match i with
  | None ⇒ x
  | Some i ⇒ v i
  end.

Fixpoint vcat {A: Type} {m : nat} : ∀ {n} (v : vec m A) (u: vec n A), vec (m+n) A.
Proof.
  destruct m.
  - intros n v u. exact u.
  - intros n v u. cbn. unfold vec in ×. apply vcons.
    + exact (v None).
    + unfold vec in ×. apply vcat.
      × intros z. exact (v (Some z)).
      × exact u.
Defined.

Notation vcomp v f := (funcomp f v).

Vector properties
Lemma vcons_comp (A: Type) B {m} (s: A) (v: vec m A) (f: A → B ) :
  vcomp (vcons s v) f = vcons (f s) (vcomp v f).
Proof.
  fext. intros [x|]. reflexivity. simpl. reflexivity.
Qed.

Lemma vcat_first {A} {m n} (v : vec m A) (u : vec n A) z:
  (vcat v u) (vindex_expand z) = v z.
Proof.
 induction m.
  - inversion z.
  - destruct z.
    + simpl. simpl. now rewrite IHm.
    + reflexivity.
Qed.

Lemma vcat_second A m n (v : vec m A) (u : vec n A) z :
  vcat v u (vindex_add m z) = u z.
Proof. induction m; cbn; eauto. Qed.

Lemma vcat_comp' A B (m n:nat) (v : vec m A) (u: vec n A) (f : A → B) x:
 f ( (vcat v u) x) = vcat (vcomp v f) (vcomp u f) x.
Proof.
  simpl in ×.
  destruct (destruct_vindex x) as [[x' ->]|[x' ->]].
  - now rewrite !vcat_first.
  - now rewrite !vcat_second.
Qed.

Lemma vcat_comp {A B} {m:nat} {n} {v : vec m A} {u : vec n A} {f : A → B} :
 vcomp (vcat v u) f = vcat (vcomp v f) (vcomp u f).
Proof. fext. intros z. unfold_funcomp. apply vcat_comp'. Qed.
