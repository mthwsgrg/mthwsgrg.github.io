Section tm.
Inductive tm (ptm : scope) : Type :=
  | var_tm : (fin) (ptm) → tm (ptm)
  | app : ∀ (n : nat), tm (ptm) → vec n (tm (ptm)) → tm (ptm)
  | lam : ∀ (n : nat), tm (Plus n ptm) → tm (ptm).

Lemma congr_app { n : nat } { qtm : scope } { s0 : tm (qtm) } { s1 : vec n (tm (qtm)) } { t0 : tm (qtm) } { t1 : vec n (tm (qtm)) } (H1 : s0 = t0) (H2 : s1 = t1) : app (qtm) n s0 s1 = app (qtm) n t0 t1 .
Proof. congruence. Qed.

Lemma congr_lam { n : nat } { qtm : scope } { s0 : tm (Plus n qtm) } { t0 : tm (Plus n qtm) } (H1 : s0 = t0) : lam (qtm) n s0 = lam (qtm) n t0 .
Proof. congruence. Qed.

Definition upRenList_tm_tm (n : nat) { q : scope } { p : scope } (xi : (fin) (q) → (fin) (p)) : (fin) (Plus n (q)) → (fin) (Plus n (p)) :=
  upRen_n n xi.

Fixpoint ren_tm { qtm : scope } { ptm : scope } (xitm : (fin) (qtm) → (fin) (ptm)) (s : tm (qtm)) : tm (ptm) :=
    match s return tm (ptm) with
    | var_tm (_) s ⇒ (var_tm (ptm)) (xitm s)
    | app (_) n s0 s1 ⇒ app (ptm) n ((ren_tm xitm) s0) ((vec_map (ren_tm xitm)) s1)
    | lam (_) n s0 ⇒ lam (ptm) n ((ren_tm (upRenList_tm_tm n xitm)) s0)
    end.

Definition upList_tm_tm (n : nat) { q : scope } { ptm : scope } (sigma : (fin) (q) → tm (ptm)) : (fin) (Plus n (q)) → tm (Plus n ptm) :=
  scons_n n ((funcomp) (var_tm (Plus n ptm)) (zero_n n)) ((funcomp) (ren_tm (shift_n n)) sigma).

Fixpoint subst_tm { qtm : scope } { ptm : scope } (sigmatm : (fin) (qtm) → tm (ptm)) (s : tm (qtm)) : tm (ptm) :=
    match s return tm (ptm) with
    | var_tm (_) s ⇒ sigmatm s
    | app (_) n s0 s1 ⇒ app (ptm) n ((subst_tm sigmatm) s0) ((vec_map (subst_tm sigmatm)) s1)
    | lam (_) n s0 ⇒ lam (ptm) n ((subst_tm (upList_tm_tm n sigmatm)) s0)
    end.

Definition upIdList_tm_tm { n : nat } { qtm : scope } (sigma : (fin) (qtm) → tm (qtm)) (Eq : ∀ x, sigma x = (var_tm (qtm)) x) : ∀ x, (upList_tm_tm n sigma) x = (var_tm (Plus n qtm)) x :=
  fun p ⇒ scons_n_eta (var_tm (Plus n qtm)) (fun p ⇒ (ap) (ren_tm (shift_n n)) (Eq p)) (fun p ⇒ eq_refl).

Fixpoint idSubst_tm { qtm : scope } (sigmatm : (fin) (qtm) → tm (qtm)) (Eqtm : ∀ x, sigmatm x = (var_tm (qtm)) x) (s : tm (qtm)) : subst_tm sigmatm s = s :=
    match s return subst_tm sigmatm s = s with
    | var_tm (_) s ⇒ Eqtm s
    | app (_) n s0 s1 ⇒ congr_app ((idSubst_tm sigmatm Eqtm) s0) ((vec_id (idSubst_tm sigmatm Eqtm)) s1)
    | lam (_) n s0 ⇒ congr_lam ((idSubst_tm (upList_tm_tm n sigmatm) (upIdList_tm_tm (_) Eqtm)) s0)
    end.

Definition upExtRen_list_tm_tm { n : nat } { q : scope } { p : scope } (xi : (fin) (q) → (fin) (p)) (zeta : (fin) (q) → (fin) (p)) (Eq : ∀ x, xi x = zeta x) : ∀ x, (upRenList_tm_tm n xi) x = (upRenList_tm_tm n zeta) x :=
  fun p ⇒ scons_n_congr (fun p ⇒ eq_refl) (fun p ⇒ (ap) (shift_n n) (Eq p)).

Fixpoint extRen_tm { qtm : scope } { ptm : scope } (xitm : (fin) (qtm) → (fin) (ptm)) (zetatm : (fin) (qtm) → (fin) (ptm)) (Eqtm : ∀ x, xitm x = zetatm x) (s : tm (qtm)) : ren_tm xitm s = ren_tm zetatm s :=
    match s return ren_tm xitm s = ren_tm zetatm s with
    | var_tm (_) s ⇒ (ap) (var_tm (ptm)) (Eqtm s)
    | app (_) n s0 s1 ⇒ congr_app ((extRen_tm xitm zetatm Eqtm) s0) ((vec_ext (extRen_tm xitm zetatm Eqtm)) s1)
    | lam (_) n s0 ⇒ congr_lam ((extRen_tm (upRenList_tm_tm n xitm) (upRenList_tm_tm n zetatm) (upExtRen_list_tm_tm (_) (_) Eqtm)) s0)
    end.

Definition upExt_list_tm_tm { n : nat } { q : scope } { ptm : scope } (sigma : (fin) (q) → tm (ptm)) (tau : (fin) (q) → tm (ptm)) (Eq : ∀ x, sigma x = tau x) : ∀ x, (upList_tm_tm n sigma) x = (upList_tm_tm n tau) x :=
  fun p ⇒ scons_n_congr (fun p ⇒ eq_refl) (fun p ⇒ (ap) (ren_tm (shift_n n)) (Eq p)).

Fixpoint ext_tm { qtm : scope } { ptm : scope } (sigmatm : (fin) (qtm) → tm (ptm)) (tautm : (fin) (qtm) → tm (ptm)) (Eqtm : ∀ x, sigmatm x = tautm x) (s : tm (qtm)) : subst_tm sigmatm s = subst_tm tautm s :=
    match s return subst_tm sigmatm s = subst_tm tautm s with
    | var_tm (_) s ⇒ Eqtm s
    | app (_) n s0 s1 ⇒ congr_app ((ext_tm sigmatm tautm Eqtm) s0) ((vec_ext (ext_tm sigmatm tautm Eqtm)) s1)
    | lam (_) n s0 ⇒ congr_lam ((ext_tm (upList_tm_tm n sigmatm) (upList_tm_tm n tautm) (upExt_list_tm_tm (_) (_) Eqtm)) s0)
    end.

Definition up_ren_ren_list_tm_tm { n : nat } { k : scope } { l : scope } { q : scope } (xi : (fin) (k) → (fin) (l)) (tau : (fin) (l) → (fin) (q)) (theta : (fin) (k) → (fin) (q)) (Eq : ∀ x, ((funcomp) tau xi) x = theta x) : ∀ x, ((funcomp) (upRenList_tm_tm n tau) (upRenList_tm_tm n xi)) x = (upRenList_tm_tm n theta) x :=
  up_ren_ren_n Eq.

Fixpoint compRenRen_tm { ktm : scope } { ltm : scope } { qtm : scope } (xitm : (fin) (qtm) → (fin) (ktm)) (zetatm : (fin) (ktm) → (fin) (ltm)) (rhotm : (fin) (qtm) → (fin) (ltm)) (Eqtm : ∀ x, ((funcomp) zetatm xitm) x = rhotm x) (s : tm (qtm)) : ren_tm zetatm (ren_tm xitm s) = ren_tm rhotm s :=
    match s return ren_tm zetatm (ren_tm xitm s) = ren_tm rhotm s with
    | var_tm (_) s ⇒ (ap) (var_tm (ltm)) (Eqtm s)
    | app (_) n s0 s1 ⇒ congr_app ((compRenRen_tm xitm zetatm rhotm Eqtm) s0) ((vec_comp (compRenRen_tm xitm zetatm rhotm Eqtm)) s1)
    | lam (_) n s0 ⇒ congr_lam ((compRenRen_tm (upRenList_tm_tm n xitm) (upRenList_tm_tm n zetatm) (upRenList_tm_tm n rhotm) (up_ren_ren_n Eqtm)) s0)
    end.

Definition up_ren_subst_list_tm_tm { n : nat } { k : scope } { l : scope } { qtm : scope } (xi : (fin) (k) → (fin) (l)) (tau : (fin) (l) → tm (qtm)) (theta : (fin) (k) → tm (qtm)) (Eq : ∀ x, ((funcomp) tau xi) x = theta x) : ∀ x, ((funcomp) (upList_tm_tm n tau) (upRenList_tm_tm n xi)) x = (upList_tm_tm n theta) x :=
  fun p ⇒ (eq_trans) (scons_n_comp' (_) (_) (_) p) (scons_n_congr (fun z ⇒ scons_n_head' (_) (_) z) (fun z ⇒ (eq_trans) (scons_n_tail' (_) (_) (xi z)) ((ap) (ren_tm (shift_n n)) (Eq z)))).

Fixpoint compRenSubst_tm { ktm : scope } { ltm : scope } { qtm : scope } (xitm : (fin) (qtm) → (fin) (ktm)) (tautm : (fin) (ktm) → tm (ltm)) (thetatm : (fin) (qtm) → tm (ltm)) (Eqtm : ∀ x, ((funcomp) tautm xitm) x = thetatm x) (s : tm (qtm)) : subst_tm tautm (ren_tm xitm s) = subst_tm thetatm s :=
    match s return subst_tm tautm (ren_tm xitm s) = subst_tm thetatm s with
    | var_tm (_) s ⇒ Eqtm s
    | app (_) n s0 s1 ⇒ congr_app ((compRenSubst_tm xitm tautm thetatm Eqtm) s0) ((vec_comp (compRenSubst_tm xitm tautm thetatm Eqtm)) s1)
    | lam (_) n s0 ⇒ congr_lam ((compRenSubst_tm (upRenList_tm_tm n xitm) (upList_tm_tm n tautm) (upList_tm_tm n thetatm) (up_ren_subst_list_tm_tm (_) (_) (_) Eqtm)) s0)
    end.

Definition up_subst_ren_list_tm_tm { n : nat } { k : scope } { ltm : scope } { qtm : scope } (sigma : (fin) (k) → tm (ltm)) (zetatm : (fin) (ltm) → (fin) (qtm)) (theta : (fin) (k) → tm (qtm)) (Eq : ∀ x, ((funcomp) (ren_tm zetatm) sigma) x = theta x) : ∀ x, ((funcomp) (ren_tm (upRenList_tm_tm n zetatm)) (upList_tm_tm n sigma)) x = (upList_tm_tm n theta) x :=
  fun p ⇒ (eq_trans) (scons_n_comp' (_) (_) (_) p) (scons_n_congr (fun x ⇒ (ap) (var_tm (Plus n qtm)) (scons_n_head' (_) (_) x)) (fun p ⇒ (eq_trans) (compRenRen_tm (shift_n n) (upRenList_tm_tm n zetatm) ((funcomp) (shift_n n) zetatm) (fun x ⇒ scons_n_tail' (_) (_) x) (sigma p)) ((eq_trans) ((eq_sym) (compRenRen_tm zetatm (shift_n n) ((funcomp) (shift_n n) zetatm) (fun x ⇒ eq_refl) (sigma p))) ((ap) (ren_tm (shift_n n)) (Eq p))))).

Fixpoint compSubstRen_tm { ktm : scope } { ltm : scope } { qtm : scope } (sigmatm : (fin) (qtm) → tm (ktm)) (zetatm : (fin) (ktm) → (fin) (ltm)) (thetatm : (fin) (qtm) → tm (ltm)) (Eqtm : ∀ x, ((funcomp) (ren_tm zetatm) sigmatm) x = thetatm x) (s : tm (qtm)) : ren_tm zetatm (subst_tm sigmatm s) = subst_tm thetatm s :=

Definition up_subst_subst_list_tm_tm { n : nat } { k : scope } { ltm : scope } { qtm : scope } (sigma : (fin) (k) → tm (ltm)) (tautm : (fin) (ltm) → tm (qtm)) (theta : (fin) (k) → tm (qtm)) (Eq : ∀ x, ((funcomp) (subst_tm tautm) sigma) x = theta x) : ∀ x, ((funcomp) (subst_tm (upList_tm_tm n tautm)) (upList_tm_tm n sigma)) x = (upList_tm_tm n theta) x :=
