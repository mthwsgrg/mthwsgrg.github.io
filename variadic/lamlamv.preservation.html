From lamlamv Require Export rules.

Lemma ty_var' (p:scope) (x : fin p) A Γ:
  A = Γ x → has_type Γ (var_tm x) A.
Proof. intros. subst. constructor. Qed.

Definition ltc {p q:scope} (Γ': ctx p) (Γ: ctx q) ξ := ∀ x, Γ (ξ x) = Γ' x.

Context renaming lemma
Lemma typing_ren (p q:scope) (Γ': ctx p) (Γ: ctx q) (ξ: fin p → fin q) (s: tm p) A :
  ltc Γ' Γ ξ → Γ' |- s : A → Γ |- s⟨ξ⟩ : A.
Proof.
  intros C H. revert q Γ ξ C. induction H; intros; asimpl; eauto.
  - unfold ltc in C. rewrite <- C. constructor.
  - constructor. apply IHhas_type. intros x.
    destruct x.
     + reflexivity.
     + asimpl. eauto.

  - constructor. apply IHhas_type. intros x. destruct x.
    × asimpl. eauto.
    × reflexivity.
Qed.

Context morphism lemma
Lemma typing_inst (p q:scope) (Γ': ctx p) (Γ: ctx q) (σ: fin p → tm q) (s: tm p) A :
  (∀ x, Γ |- σ x : Γ' x) → Γ' |- s : A → Γ |- s[σ] : A.
Proof.
Proof.
  intros C H. revert q Γ σ C. induction H; intros; asimpl; eauto.
  - constructor. apply IHhas_type. intros x.
     destruct x. asimpl.
     + constructor.
     + eapply typing_ren; eauto. intros x. now asimpl.
  - constructor. apply IHhas_type. intros x; destruct x.
    + simpl. eapply typing_ren; eauto. intros. unfold ltc. now asimpl.
    + asimpl. constructor.
Qed.

Preservation

Lemma step_typing (p:scope) (Γ: ctx p) s A :
  Γ |- s : A → ∀ t, step s t → Γ |- t : A.
Proof.
  induction 1; intros; cbn.
  - inv H.
  - inv H0. econstructor. now apply IHhas_type.
  - inv H2.
    + inv H. eapply typing_inst; try eassumption.
      intros x. destruct x; asimpl;simpl; eauto.
apply ty_var'; eauto.
    + simpl. eapply ty_appv; eauto.
  - inv H0. econstructor. now apply IHhas_type.
  - inv H1.
    + inv H. eapply typing_inst; try eassumption.
      intros x.
      destruct x; asimpl;simpl; eauto.
    + eapply ty_app; eauto.
    + eapply ty_app; eauto.
Qed.
