Reduction rules

Inductive step {p:scope} : tm p → tm p → Prop :=
| step_betav (n: nat) b (t: vec n (tm p) ) : step (appv _ (lamv n b) t) (subst1 (scons_n n t ids) b)
| step_absv (n: nat) b1 b2 : @step (n ± p) b1 b2 → step (lamv n b1) (lamv n b2)
| step_appvL (n: nat) s1 s2 t : step s1 s2 → step (appv n s1 t) (appv _ s2 t)
| step_beta b t : step (app (lam b) t) b[t..]
| step_abs b1 b2 : @step (Sp p) b1 b2 → step (lam b1) (lam b2)
| step_appL s1 s2 t : step s1 s2 → step (app s1 t) (app s2 t)
| step_appR s t1 t2 : step t1 t2 → step (app s t1) (app s t2).

#[export] Hint Constructors step.

Lemma step_betav' (p:scope) (n: nat) b (tBar: vec n (tm p)) t':
  t' = b[scons_n n tBar ids] → step (appv _ (lamv n b) tBar) t'.
Proof. intros →. now constructor. Qed.

Lemma step_beta' p b (t t': tm p):
  t' = b[t..] → step (app (lam b) t) t'.
Proof. intros →. now constructor. Qed.

Multi-step reduction

Lemma mstep_lamv (n: nat) (p:scope) (s t : tm (n ± p)) :
  star step s t → star step (lamv n s) (lamv n t).
Proof. induction 1; eauto. Qed.

Lemma mstep_lam p (s t : tm (Sp p)) :
  star step s t → star step (lam s) (lam t).
Proof. induction 1; eauto. Qed.

Lemma mstep_app p (s1 s2 : tm p) (t1 t2 : tm p) :
  star step s1 s2 → star step t1 t2 → star step (app s1 t1) (app s2 t2).
Proof with eauto.
  intros ms. induction 1. induction ms... auto...
Qed.

Substitutivity of reduction

Lemma step_inst {p q:scope} (f : fin p → tm q) (s t : tm p) :
  step s t → step (subst_tm f s) (subst_tm f t).
Proof.
  intros st. revert q f. induction st; intros; cbn.
  - apply step_betav'. now asimpl.
  - apply step_absv. eapply IHst.
  - apply step_appvL, IHst.
  - apply step_beta'. now asimpl.
  - apply step_abs. eapply IHst.
  - apply step_appL, IHst.
  - apply step_appR, IHst.
Qed.

Lemma mstep_inst (p q:scope) (f : fin p → tm q) (s t : tm p) :
  star step s t → star step (subst1 f s) (subst1 f t).
Proof. induction 1; eauto using step_inst. Qed.

Lemma step_naturality (p q:scope) (s: tm p) (ξ: fin p → fin q) s' :
  step (ren_tm ξ s) s' → ∃ s'', step s s'' ∧ s' = (ren_tm ξ s'').
Proof.
  intros H.
  dependent induction H.
  - destruct s; try inversion x. subst.

    apply inj_pair2_eq_dec in H2; try (apply eq_nat_dec).
    rewrite H2 in x.
    destruct s; try inversion H1. subst.
    simpl in H1.
    rewrite H1 in x.
    apply inj_pair2_eq_dec in H3; try (apply eq_nat_dec).
    rewrite H3.
    ∃ (subst_tm (scons_n n v ids) s). split. eauto.
    asimpl. reflexivity.

  - destruct s; try inversion x. subst.
    apply inj_pair2_eq_dec in H2; try (apply eq_nat_dec).
    rewrite H2 in H.
    rewrite H2 in IHstep.
    edestruct (IHstep _ s) as (?&?&?);[reflexivity|].
    subst. eexists;eauto.

  - destruct s; inversion x. subst.
    apply inj_pair2_eq_dec in H3; try (apply eq_nat_dec).
    edestruct (IHstep _ s) as (?&?&?); [reflexivity|].
    subst. eexists; eauto.

  - destruct s; try inversion x. subst.
    destruct s1; try inversion H0. subst.
    ∃ (subst_tm (s2..) s1). split. eauto.
    asimpl. reflexivity.
  - destruct s; inversion x. subst.
    edestruct (IHstep _ s) as (?&?&?); [reflexivity|].
    subst. eexists; eauto.
  - destruct s; inversion x. subst.
    edestruct (IHstep _ s3) as (?&?&?); [reflexivity|].
    subst. eexists; eauto.
  - destruct s; inversion x. subst.
    edestruct (IHstep _ s2) as (?&?&?); [reflexivity|].
    subst. eexists; eauto.
Qed.

Types

Inductive ty : Type :=
  | B : ty
  | arr n : (vec n ty) → ty → ty
  | Fun : ty → ty → ty .

Print ty_ind.

Definition ctx (p:scope) := fin p → ty.

Typing rules
Inductive has_type {p:scope} (Γ : ctx p) : tm p → ty → Prop :=
| ty_var (x : fin p) :
    has_type Γ (var_tm x) (Γ x)
| ty_absv (n:nat) (ABar : vec n ty) ( A : ty) (s : tm (n ± p)) :
    @has_type (n ± p) (scons_n n ABar Γ) s A →
    has_type Γ (lamv n s) (arr ABar A)
| ty_appv (n:nat) (ABar : vec n ty) ( B : ty) (s : tm p) tBar :
    has_type Γ s (arr ABar B) →
    (∀ x, has_type Γ (tBar x) (ABar x)) →
    has_type Γ (appv n s tBar) B
| ty_abs (A B : ty) (s : tm (Sp p)) :
    @has_type (Sp p) (A .: Γ) s B →
    has_type Γ (lam s) (Fun A B)
| ty_app (A B : ty) (s t : tm p) :
    has_type Γ s (Fun A B) →
    has_type Γ t A →
    has_type Γ (app s t) B.

Notation "Gamma |- s : A" := (has_type Gamma s A) (at level 20, s at level 99).

#[export] Hint Constructors has_type.

Lemma ty_var' (p:scope) (x : fin p) A Γ:
  A = Γ x → has_type Γ (var_tm x) A.
Proof. intros. subst. constructor. Qed.

Definition ltc {p p':scope} (Γ: ctx p) (Δ: ctx p') ξ := ∀ x, Δ (ξ x) = Γ x.

Context renaming lemma
Lemma typing_ren (p q:scope) (Γ: ctx p) (Δ: ctx q) (ξ: fin p → fin q) (s: tm p) A :
  ltc Γ Δ ξ → Γ |- s : A → Δ |- (ren1 ξ s) : A.
Proof.
  intros C H. revert q Δ ξ C. induction H; intros; asimpl; eauto.
  - unfold ltc in C. rewrite <- C. constructor.
  - constructor. apply IHhas_type. intros x.
    destruct x.
     + reflexivity.
     + asimpl. eauto.

  - constructor. apply IHhas_type. intros x. destruct x.
    × asimpl. eauto.
    × reflexivity.
Qed.

Context morphism lemma

Lemma typing_inst (p q:scope) (Γ: ctx p) (Δ: ctx q) (σ: fin p → tm q) (s: tm p) A :
  (∀ x, Δ |- σ x : Γ x) → Γ |- s : A → Δ |- (subst1 σ s) : A.
Proof.
Proof.
  intros C H. revert q Δ σ C. induction H; intros; asimpl; eauto.
  - constructor. apply IHhas_type. intros x.
     destruct x. asimpl.
     + constructor.
     + eapply typing_ren; eauto. intros x. now asimpl.

  - constructor. apply IHhas_type. intros x; destruct x.
    + simpl. eapply typing_ren; eauto. intros. unfold ltc. now asimpl.
    + asimpl. constructor.

Qed.

Preservation

Lemma step_typing (p:scope) (Γ: ctx p) s A :
  Γ |- s : A → ∀ s', step s s' → Γ |- s' : A.
Proof.
  induction 1; intros; cbn.
  - inv H.
  - inv H0. econstructor. now apply IHhas_type.
  - inv H2.
    + inv H. eapply typing_inst; try eassumption.
      intros x. destruct x; asimpl;simpl; eauto.
apply ty_var'; eauto.
    + simpl. eapply ty_appv; eauto.
  - inv H0. econstructor. now apply IHhas_type.
  - inv H1.
    + inv H. eapply typing_inst; try eassumption.
      intros x.
      destruct x; asimpl;simpl; eauto.
    + eapply ty_app; eauto.
    + eapply ty_app; eauto.
Qed.
